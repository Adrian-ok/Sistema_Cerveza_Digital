
python = 3.11

node >= 22.11.0

Levantar servidor backend:
- abra una terminal dentro de la carpeta "Sistema_cerveza_digital"
- ubiquese dentro de "backend/"
- crear entorno virtual "python -m venv venv"
- ubiquese dentro de "backend/venv/scripts"
- active el entorno virtual "activate.bat"
- instalar los requrimientos "pip install requirements.txt"
- ubiquese dentro de "backend/cerveza_digital/"
- aplicar las migraciones "python manage.py migrate"
- crear el super usuario "python manage.py createsuperuser"
- levantar el servidor backend "python manage.py runserver"
- acceder a la documentacion de la API "url_local/docs"
- acceder al panel de administracion "url_local/admin"
  - accceda con los datos de su super usuario

Levantar servidor frontend:
- abra una terminal dentro de la carpeta "Sistema_cerveza_digital"
- ubiquese dentro de "frontend/cerveza_digital"
- instale las dependencias "npm install"
- levantar el servidor "npm run dev"
- acceder a la web 
  - cliente: "url_local/"
  - operario: "url_local/admin"
    - accede con los datos de su super usuario
